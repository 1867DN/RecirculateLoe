    <h2>Formulario de Contacto</h2>
    <form aria-label="Formulario de contacto" id="contactForm">
      <div>
        <label for="nombre">Nombre</label>
        <input type="text" id="nombre" placeholder="Ingresa tu nombre" aria-required="true">
        <div class="error" id="nombreError">Este campo es obligatorio</div>
      </div>

      <div>
        <label for="email">Correo</label>
        <input type="email" id="email" placeholder="ejemplo@email.com" aria-required="true">
        <div class="error" id="emailError">Ingrese un correo válido</div>
      </div>

      <div style="grid-column: span 2;">
        <label for="mensaje">Mensaje</label>
        <textarea id="mensaje" placeholder="Escribe tu mensaje aquí..." aria-required="true"></textarea>
        <div class="error" id="mensajeError">Este campo es obligatorio</div>
      </div>

      <button type="submit">Enviar</button>
      <div class="success" id="formSuccess">Gracias por tu mensaje! Pronto nos pondremos en contacto contigo</div>
    </form>

  </div>

  <script>
    // validación de elementos
    const form = document.getElementById('contactForm');
    const nombre = document.getElementById('nombre');
    const email = document.getElementById('email');
    const mensaje = document.getElementById('mensaje');

    const nombreError = document.getElementById('nombreError');
    const emailError = document.getElementById('emailError');
    const mensajeError = document.getElementById('mensajeError');
    const formSuccess = document.getElementById('formSuccess');

    form.addEventListener('submit', (e) => {
      e.preventDefault();
      let valid = true;

      // Validar nombre
      if (nombre.value.trim() === "") {
        nombreError.style.display = "block";
        valid = false;
      } else {
        nombreError.style.display = "none";
      }

      // Validar email
      if (!email.value.includes("@")) {
        emailError.style.display = "block";
        valid = false;
      } else {
        emailError.style.display = "none";
      }

      // Validar mensaje
      if (mensaje.value.trim() === "") {
        mensajeError.style.display = "block";
        valid = false;
      } else {
        mensajeError.style.display = "none";
      }

      // Si todo está bien
      if (valid) {
        formSuccess.style.display = "block";
        form.reset();
        setTimeout(() => formSuccess.style.display = "none", 3000);
      }
    });
  </script>
